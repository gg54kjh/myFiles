<script>
var table = document.querySelector('#branah-keyboard'),
    tds = table.querySelectorAll('.branah-key'),
    ed = document.querySelector('#editor'),
    flag = 0, FLAG = 0;

// --------  Кнопки с буквами и цифрами  ----------
for(i=0; i<tds.length; i++){
   if(tds[i].hasAttribute('id') &&  (tds[i].id!='b1'&&tds[i].id!='b2'))    tds[i].onclick=click_;
}
function click_(){ 
     // flag==1 ? letter = this.textContent.toUpperCase() : letter = this.textContent;
     if(flag==0)
        var  letter = this.querySelector('.branah-label-natural').textContent 
     else  letter=this.querySelector('.branah-label-reference').textContent;
     if(FLAG==1) letter = letter.toUpperCase();
     start = ed.selectionStart;   end = ed.selectionEnd; 
     doo = ed.value.substring(0, ed.selectionStart); 
     posle = ed.value.substring(ed.selectionEnd, ed.value.length); 
   
     if(ed.getAttribute('dir') == 'rtl') {
         letter=letter.replace(/\s/, ''); 
         if(start==end){
             ed.value=doo +letter +posle;  ed.selectionStart = ed.selectionEnd = end +1
         }
         else{
             ed.value=doo  +letter+  posle; ed.selectionStart = ed.selectionEnd = start +1
         }
     }
     if(ed.getAttribute('dir') == 'ltr') {
         if(start==end){
             ed.value=doo+ letter +posle; ed.selectionEnd = end +1
         }
        else{
             ed.value=doo+ letter +posle; ed.selectionEnd = start +1
        }
     }
      ed.focus(); 
}

// --------  Кнопка  Backspace  ---------- 
table.querySelector('#branah-backspace').onclick=function(){
    var value = ed.value;
    var pos = ed.selectionStart - 1,
          doo = value.substring(0, ed.selectionStart - 1),
          posle = value.substring(ed.selectionEnd);
    ed.value = doo+posle; 
    ed.selectionStart = ed.selectionEnd = pos;
    ed.focus();
}

// --------  Кнопка  Enter  ---------- 
table.querySelector('#branah-enter').onclick=function(){
   var doo = ed.value.substring(0, ed.selectionStart),
      end = ed.selectionEnd,
      posle = ed.value.substring(end);
   ed.value = doo +'\n'+ posle;
   ed.selectionStart = ed.selectionEnd = end+1;
   ed.focus();
}

// --------  Кнопка  левый Shift  ---------- 
table.querySelector('#branah-left-shift').onclick=function(){
     if(this.className=='pressed'){
          flag = 0;  this.classList.remove('pressed'); 
          document.querySelector('#hideletters').innerHTML=
          `.branah-label-reference{display:none}` 
     }
     else {
          flag = 1;  this.className = 'pressed';
          document.querySelector('#hideletters').innerHTML=
         `.branah-label-natural{display:none} .branah-label-reference{text-align:center !important}` 
    }
      ed.focus();
}

// --------  Кнопка  левый Ctrl  ---------- 
table.querySelector('#branah-left-ctrl').onclick=function(){
   if(this.className=='pressed'){
       this.classList.remove('pressed');  ed.setAttribute('dir', 'rtl')
   }
   else {
       this.className = 'pressed';  ed.setAttribute('dir', 'ltr') 
    }
}

// --------  Кнопка  Caps  ----------
table.querySelector('#branah-caps-lock').onclick=function(){
   if(this.className=='pressed'){
       this.classList.remove('pressed');   FLAG = 0
   }
   else {
       this.className = 'pressed';   FLAG = 1
   }
     ed.focus();  
}

// --------  Кнопка  Clear  All  ----------
table.querySelector('#b2').onclick=function(){
    ed.value = '';   ed.focus();
}

</script>
